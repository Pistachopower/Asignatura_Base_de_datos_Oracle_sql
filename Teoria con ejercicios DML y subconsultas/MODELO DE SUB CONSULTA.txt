/*

*/
--V_EDAD_MEDIA= 
SELECT AVG(EDAD)
FROM ALUMNOS;

/PREPARO LA CONSULTA GENERAL/
SELECT * 
FROM ALUMNOS
WHERE EDAD>= V_EDAD_MEDIA-2 
AND EDAD <= V_EDAD_MEDIA+2; 


/UNE LA CONSULTA FINAL/
SELECT * 
FROM ALUMNOS
WHERE EDAD>= (SELECT AVG(EDAD)
                FROM ALUMNOS)-2 
AND EDAD <= (SELECT AVG(EDAD)
FROM ALUMNOS)+2;
