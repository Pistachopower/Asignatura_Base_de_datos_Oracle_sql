SELECT NOMBRE, EDAD
FROM ALUMNOS
WHERE EDAD IS NULL;

SELECT NOMBRE,EDAD, EDAD +1 
FROM ALUMNO
WHERE EDAD IS NOT NULL;

SELECT NOMBRE, EDAD, NVL(EDAD, 27) AS NUEVA_EDAD --, NVL(EDAD, 'SIN EDAD')
   , FECNAC , NVL(FECNAC, SYSDATE), NVL(FECNAC, TO_DATE('01/01/2022','DD/MM/YYYY'))
FROM ALUMNOS;

SELECT NOMBRE, SYSDATE - FECNAC, ROUND(SYSDATE - FECNAC,2) AS DIAS_VIVIDOS, TRUNC(SYSDATE- FECNAC),CEIL(SYSDATE- FECNAC)
FROM ALUMNOS;


SELECT NOMBRE, CONCAT(CONCAT(NOMBRE,''),APELLIDOS) ,NOMBRE ||' '|| APELLIDOS AS NOMBRE_COMPLETO
FROM ALUMNOS;

FROM ALUMNOS
WHERE SUBSTR(NOMBRE,1,1) = 'A' OR SUBSTR(NOMBRE,1,1) = 'a';

-- SEPARA LOS APELLIDOS CADA UNO POR SU LADO
SELECT APELLIDOS,INSTR(APELLIDOS,' ',1) AS POS_ESPACIO,
       --SUBSTR(APELLIDOS,1, POS_ESPACIO), SUBSTR(APELLIDOS, POS_ESPACIO)
       SUBSTR(APELLIDOS,1, INSTR(APELLIDOS,' ',1)), SUBSTR(APELLIDOS, INSTR(APELLIDOS,' ',1))
FROM ALUMNOS;


-- FUNCIONES DE FECHA
SELECT SYSDATE,ADD_MONTHS(SYSDATE,1), ADD_MONTHS(SYSDATE,12*5), LAST_DAY(SYSDATE), 
MONTHS_BETWEEN(SYSDATE, TO_DATE('01/01/2000','DD/MM/YYYY'))/12
FROM DUAL;


SELECT NOMBRE, APELLIDOS,FECNAC, TRUNC(MONTHS_BETWEEN(SYSDATE, FECNAC)/12) AS EDAD_CALCULADA
FROM ALUMNOS;

SELECT TO_CHAR(SYSDATE,'DD'),TO_CHAR(SYSDATE,'MM'),TO_CHAR(SYSDATE,'YYYY'),TO_CHAR(SYSDATE,'ww'),TO_CHAR(SYSDATE,'DAY')
