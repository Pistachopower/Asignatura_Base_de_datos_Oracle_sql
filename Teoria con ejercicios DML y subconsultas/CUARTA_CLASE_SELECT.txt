-- MOSTRAR EL NOMBRE Y EL DIA DE LAS SEMANA (LUNES, MARTES...ETC) QUE ERA EN EL DECIMO CUMPLEAÃ‘OS DE CADA ALUMNO
SELECT UPPER(NOMBRE ||' '||APELLIDOS), FECNAC, ADD_MONTHS(FECNAC, 12*10) AS FECHA_DECIMO_CUMPLE,
       TO_CHAR(ADD_MONTHS(FECNAC, 12*10),'DAY') , TO_CHAR(FECNAC,'DAY')
FROM ALUMNOS
WHERE UPPER(CURSO) ='1DAW';


-- PUEDO PONER VALORES FIJOS
SELECT NOMBRE, FECNAC, '           ' AS ESPACIO,'MOSTRA LO QUE ME DA LA GANA' AS UNA_CADENA, 33 AS UN_NUMERO, TO_DATE('01/01/2003','DD/MM/YYYY') AS UNA_FECHA
FROM ALUMNOS
WHERE UPPER(CURSO) ='1DAW';


SELECT UPPER(NOMBRE ||' '||APELLIDOS), FECNAC, TRUNC(MONTHS_BETWEEN(SYSDATE, FECNAC)/12) AS EDAD_CALCULADA
FROM ALUMNOS
WHERE UPPER(CURSO) ='1DAW';

-- QUIEN MIENTE EN LA EDAD
SELECT UPPER(NOMBRE ||' '||APELLIDOS), FECNAC, TRUNC(MONTHS_BETWEEN(SYSDATE, FECNAC)/12) AS EDAD_CALCULADA, EDAD
FROM ALUMNOS
WHERE TRUNC(MONTHS_BETWEEN(SYSDATE, FECNAC)/12) <> EDAD;
